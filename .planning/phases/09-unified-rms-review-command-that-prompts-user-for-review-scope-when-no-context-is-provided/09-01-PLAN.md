**Step 3 — Add the new `review` sub-command**:
```typescript
program
  .command('review')
  .description('Run a code review — prompts for scope if not provided')
  .argument('[scope]', 'Review scope: "local" for git diff, "pr" for GitHub PR')
  .argument('[pr-number]', 'GitHub PR number (required when scope is "pr")')
  .option('--focus <area>', 'Focus area (e.g., security, performance)')
  .action(async (scope: string | undefined, prArg: string | undefined, opts: { focus?: string }) => {
    const projectRoot = process.cwd();

    if (!scope) {
      // No scope provided — output structured prompt for host AI agent to present to user
      console.log([
        '[rms] What would you like to review?',
        '',
        '  1. local  — Review staged and unstaged git changes in the working tree',
        '  2. pr     — Review a GitHub Pull Request (you will need to provide the PR number)',
        '',
        'Re-invoke with your choice:',
        '  rms review local [--focus <area>]',
        '  rms review pr <pr-number> [--focus <area>]',
      ].join('\n'));
      return;
    }

    if (scope === 'local') {
      await runLocalReview({ projectRoot, focus: opts.focus });
      return;
    }

    if (scope === 'pr') {
      if (!prArg) {
        console.error('[rms] PR number is required. Usage: rms review pr <pr-number> [--focus <area>]');
        process.exit(1);
      }
      const prNumber = parseInt(prArg, 10);
      if (isNaN(prNumber) || prNumber <= 0) {
        console.error(`Invalid PR number: "${prArg}". Must be a positive integer.`);
        process.exit(1);
      }
      await runPrReview({ projectRoot, prNumber, focus: opts.focus });
      return;
    }

    console.error(`[rms] Unknown scope "${scope}". Use "local" or "pr".`);
    process.exit(1);
  });
```
