    export async function resolveCopilotToken(): Promise<string> {
      // Priority 1: GITHUB_TOKEN env var (same token used for PR diffs)
      if (process.env['GITHUB_TOKEN']) return process.env['GITHUB_TOKEN'];

      // Priority 2: opencode auth.json
      const authPath = join(homedir(), '.local', 'share', 'opencode', 'auth.json');
      try {
        const raw = await readFile(authPath, 'utf8');
        const auth = JSON.parse(raw) as Record<string, unknown>;
        const copilotAuth = auth['github-copilot'] as Record<string, unknown> | undefined;
        const token = copilotAuth?.['access'];
        if (typeof token === 'string' && token.length > 0) return token;
      } catch {
        // File not found or unreadable — fall through to error
      }

      throw new Error(
        '[rms] copilot provider requires a GitHub token.\n' +
        '  Set GITHUB_TOKEN or authenticate with OpenCode (opencode auth login).',
      );
    }
    ```

    function parseSpec(raw: string, label: string): AgentModelSpec {
      // Accept raw "github-copilot/model-id" from `opencode models` output
      if (raw.startsWith('github-copilot/')) {
        const model = raw.slice('github-copilot/'.length);
        if (!model) {
          console.error(`Invalid ${label} spec "${raw}": model ID cannot be empty`);
          process.exit(1);
        }
        return { provider: 'copilot', model };
      }

      // Standard "provider:model-id" format
      const colonIdx = raw.indexOf(':');
      if (colonIdx === -1) {
        console.error(
          `Invalid ${label} spec "${raw}". Expected format: provider:model (e.g. anthropic:claude-opus-4-5) or github-copilot/model-id`,
        );
        process.exit(1);
      }
      const provider = raw.slice(0, colonIdx) as AgentModelSpec['provider'];
      const model = raw.slice(colonIdx + 1);
      if (!['openai', 'anthropic', 'google', 'copilot'].includes(provider)) {
        console.error(
          `Invalid provider "${provider}". Must be one of: openai, anthropic, google, copilot`,
        );
        process.exit(1);
      }
      if (!model) {
        console.error(`Invalid ${label} spec "${raw}": model ID cannot be empty`);
        process.exit(1);
      }
      return { provider, model };
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| env → config.ts | GITHUB_TOKEN read from process.env — attacker with env access can supply arbitrary token |
| filesystem → config.ts | auth.json read from ~/.local/share/opencode/auth.json — file is user-controlled |
| user CLI input → parseSpec | Raw spec strings from CLI flags (--reviewer, --validator, --writer) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-16-01 | Information Disclosure | resolveCopilotToken() | accept | Token is read only from user's own env/files; not logged or emitted to stdout. Low risk: attacker with local access already has the token. |
| T-16-02 | Spoofing | parseSpec() | mitigate | Provider whitelist enforced: only `['openai','anthropic','google','copilot']` accepted. Unknown providers exit 1. |
| T-16-03 | Tampering | auth.json read | accept | File is user-owned and not integrity-checked. Risk: compromised auth.json supplies bad token → API rejects it at call time with a clear 401. No sensitive data amplification. |
| T-16-04 | Denial of Service | resolveCopilotToken() throws | mitigate | Error path exits 1 with actionable message rather than hanging or silently using a null token. |
</threat_model>
