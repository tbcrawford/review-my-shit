```typescript
export interface AgentModelSpec {
  model: string;       // e.g. "github-copilot/claude-opus-4.6"
  variant?: "high_thinking" | "no_thinking";

  reviewer:  AgentModelSpec;
  validator: AgentModelSpec;
  writer:    AgentModelSpec;
}

export interface RmsConfig {
  opencode: EditorAgentConfig;
  cursor:   EditorAgentConfig;
}

export interface FlatRmsConfig {
  reviewer:  { provider: string; model: string };
  validator: { provider: string; model: string };
  writer:    { provider: string; model: string };
}

export { RmsConfigSchema, FlatRmsConfigSchema, AgentModelSpecSchema };
```

  <behavior>
    **loadRmsConfig:**
    - Parse raw JSON as `unknown`
    - First try `RmsConfigSchema.safeParse(parsed)` — if valid, return it
    - Then try `FlatRmsConfigSchema.safeParse(parsed)` — if valid, migrate: map flat→nested
    - Flat→nested migration rule: discard `provider`, keep `model`, no variant
      ```
      opencode: { reviewer: {model: flat.reviewer.model}, validator: ..., writer: ... }
      cursor:   { reviewer: {model: flat.reviewer.model}, validator: ..., writer: ... }
      ```
      (Both editors get the same models from the old flat config — best-effort migration)
    - If neither schema matches, throw "Invalid rms config" as before
    - Returns `RmsConfig | null` (null if file absent)

    **DEFAULT_RMS_CONFIG:**
    Target defaults (from phase context):
    ```typescript
    export const DEFAULT_RMS_CONFIG: RmsConfig = {
      opencode: {

        validator: { model: 'github-copilot/gpt-5.4',          variant: 'high_thinking' },
        writer:    { model: 'github-copilot/claude-haiku-4.5', variant: 'no_thinking'   },

      },
    ```

    - New signature: `resolveAgentModel(spec: AgentModelSpec): Promise<LanguageModel>`

    - When `spec.variant` is undefined: no `providerOptions` (let the API decide)
    - Keep `resolveCopilotToken()` unchanged
    - REMOVE the provider switch (anthropic/google/openai branches) — all models go through copilot now

    **resolveModels() helper NOTE:** This function lives in index.ts, NOT config.ts. Config.ts only exports resolveAgentModel. The index.ts update is Plan 03.
