<interfaces>
<!-- New CLI interface from Plan 03 that templates invoke -->

rms settings (new flags):
  --reviewer <model[:variant]>    set reviewer model (both editors by default)
  --validator <model[:variant]>   set validator model (both editors by default)
  --writer <model[:variant]>      set writer model (both editors by default)
  --opencode                      scope to opencode section only
  --cursor                        scope to cursor section only
  --reset                         delete config

rms settings (display, no flags):
  Shows both opencode and cursor sections:
    opencode:
      reviewer   github-copilot/claude-opus-4.6 [high_thinking]
      ...
    cursor:
      reviewer   claude-4.6-opus-high-thinking
      ...

Spec string format: "model" or "model:variant"
  Valid variants: high_thinking, no_thinking
  Examples:
    github-copilot/claude-opus-4.6:high_thinking
    github-copilot/gpt-5.4:high_thinking
    github-copilot/claude-haiku-4.5:no_thinking
    claude-4.6-opus-high-thinking          (cursor model — no variant)
    gpt-5.4-high                           (cursor model — no variant)
    gpt-5.4-mini-none                      (cursor model — no variant)
